## Executive Summary

MDT v2.0 is a real-time, AI-powered smart transit planner for Mumbai's chaotic multi-modal network. It combines:

- **Graph-based multi-modal routing** (Local Train, Metro, Mono, Bus, Auto, Ferry)
- **MDT Sathi** — a hyper-local AI commuter assistant who speaks like a Mumbaikar
- **Crisis simulation** — judges can see the platform respond live to rain/strikes/block
- **Offline-first architecture** — works even when venue Wi-Fi collapses
- **7+ winning differentiators** not in the original plan

**Target Impact:** Serve 8+ million daily Mumbai commuters who lose an average of 47 minutes/day to unplanned disruptions.

### Data Flow

```
User Input (Origin → Destination)
    ↓
Graph Lookup (check localStorage first)
    ↓
Apply Active Disruptions (filter blocked edges)
    ↓
Dijkstra Algorithm (with multi-modal penalties)
    ↓
Route Object (segments, time, cost, CO₂)
    ↓
Sathi Engine (generate localized commentary)
    ↓
Map Render (Leaflet polylines per mode)
    ↓
UI Update (Route card + Sathi message + Stats)
```

// /src/data/mumbaiGraph.js

const GRAPH_CACHE_KEY = "mdt_transit_graph_v2";
const CACHE_VERSION = "2.0.1";

export function initializeOfflineData() {
  const cached = localStorage.getItem(GRAPH_CACHE_KEY);
  
  if (cached) {
    const parsed = JSON.parse(cached);
    if (parsed.version === CACHE_VERSION) {
      console.log("✅ MDT: Offline graph loaded from cache");
      return parsed.data;
    }
  }
  
  // First load: save to cache
  const graphData = { nodes: NODES, edges: EDGES };
  localStorage.setItem(GRAPH_CACHE_KEY, JSON.stringify({
    version: CACHE_VERSION,
    timestamp: Date.now(),
    data: graphData
  }));
  
  console.log("📶 MDT: Graph cached for offline use");
  return graphData;
}

export function buildAdjacencyList(edges) {
  const adj = {};
  edges.filter(e => e.active).forEach(edge => {
    if (!adj[edge.from]) adj[edge.from] = [];
    if (!adj[edge.to]) adj[edge.to] = [];
    adj[edge.from].push(edge);
    // Edges are bidirectional for most modes
    if (edge.mode !== "ferry") {
      adj[edge.to].push({ ...edge, from: edge.to, to: edge.from });
    }
  });
  return adj;
}

## Routing Engine

### File: `/src/logic/router.js`

#### Multi-Modal Dijkstra

```javascript
/**
 * Multi-Modal Dijkstra with:
 * - Mode transfer penalties (e.g., switching Train → Metro costs 5 mins)
 * - Peak hour multipliers
 * - Disruption-aware edge filtering
 * - Up to 3 alternative routes
 */

const TRANSFER_PENALTY = {
  same: 0,
  local_to_metro: 7,    // Walk to metro station
  local_to_bus: 5,
  metro_to_bus: 3,
  local_to_walk: 2,
  any_to_ferry: 10,
};

const MODE_COLORS = {
  local_central: "#E74C3C",   // Central Line Red
  local_western: "#3498DB",   // Western Line Blue
  local_harbour:  "#27AE60",  // Harbour Line Green
  metro1: "#00BCD4",          // Metro Cyan
  metro2a: "#9C27B0",         // Metro Purple
  metro7: "#FF9800",          // Metro Orange
  monorail: "#FF5722",        // Mono Red
  bus: "#F39C12",             // BEST Bus Yellow
  walk: "#95A5A6",            // Walk Grey
  ferry: "#1A237E",           // Ferry Navy
};

export function dijkstra({ graph, origin, destination, preferences }) {
  const { nodes, adjList } = graph;
  const { isPeakHour, optimizeFor, avoidModes = [] } = preferences;

  // Priority queue: [cost, nodeId, path, mode]
  const pq = new MinHeap();
  const dist = {};
  const prev = {};
  
  Object.keys(nodes).forEach(id => { dist[id] = Infinity; });
  dist[origin] = 0;
  pq.push({ cost: 0, node: origin, path: [], currentMode: null });

  while (!pq.isEmpty()) {
    const { cost, node, path, currentMode } = pq.pop();

    if (node === destination) {
      return buildRouteResult(path, nodes);
    }

    if (cost > dist[node]) continue;

    const neighbors = adjList[node] || [];
    
    for (const edge of neighbors) {
      if (avoidModes.includes(edge.mode)) continue;
      if (!edge.active) continue;

      // Calculate edge cost based on optimization preference
      let edgeCost = optimizeFor === "time"
        ? edge.duration * (isPeakHour ? edge.peakFactor : 1.0)
        : optimizeFor === "cost"
        ? edge.cost
        : edge.co2;

      // Add transfer penalty when switching modes
      const transferKey = currentMode
        ? `${currentMode.split('_')[0]}_to_${edge.mode.split('_')[0]}`
        : null;
      const penalty = transferKey ? (TRANSFER_PENALTY[transferKey] || 0) : 0;

      const newCost = cost + edgeCost + penalty;

      if (newCost < dist[edge.to]) {
        dist[edge.to] = newCost;
        pq.push({
          cost: newCost,
          node: edge.to,
          path: [...path, edge],
          currentMode: edge.mode
        });
      }
    }
  }

  return null; // No route found
}

function buildRouteResult(path, nodes) {
  let totalTime = 0, totalCost = 0, totalCO2 = 0;
  const segments = [];

  path.forEach((edge, i) => {
    totalTime += edge.duration;
    totalCost += edge.cost;
    totalCO2 += edge.co2;
    
    segments.push({
      from: nodes[edge.from],
      to: nodes[edge.to],
      mode: edge.mode,
      modeLabel: getModeLabel(edge.mode),
      color: MODE_COLORS[edge.mode],
      duration: edge.duration,
      cost: edge.cost,
      waypoints: edge.waypoints,
      isTransfer: i > 0 && path[i-1].mode !== edge.mode
    });
  });

  return {
    segments,
    totalTime,       // minutes
    totalCost,       // rupees
    totalCO2,        // grams
    co2Saved: calculateCO2Saved(totalCO2, path), // vs auto-rickshaw
    monsoonSafe: isMonsoonSafe(path),
    crowdLevel: getRouteCrowdLevel(path),
  };
}
```

## Offline-Ready Architecture

### File: `/src/hooks/useOffline.js`

```javascript
import { useState, useEffect } from 'react';

const CACHE_KEYS = {
  GRAPH: "mdt_transit_graph_v2",
  SEARCH_STATE: "mdt_search_state",
  DISRUPTIONS: "mdt_active_disruptions",
  SATHI_LOG: "mdt_sathi_log",
};

export function useOffline() {
  const [isOnline, setIsOnline] = useState(navigator.onLine);
  const [offlineReady, setOfflineReady] = useState(false);

  useEffect(() => {
    const handleOnline = () => setIsOnline(true);
    const handleOffline = () => setIsOnline(false);
    
    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);
    
    // Check if we have cached data
    const cached = localStorage.getItem(CACHE_KEYS.GRAPH);
    setOfflineReady(!!cached);
    
    return () => {
      window.removeEventListener('online', handleOnline);
      window.removeEventListener('offline', handleOffline);
    };
  }, []);

  const persistState = (key, data) => {
    try {
      localStorage.setItem(CACHE_KEYS[key], JSON.stringify(data));
    } catch (e) {
      console.warn("MDT: localStorage full, clearing old cache");
      localStorage.removeItem(CACHE_KEYS.SATHI_LOG); // Clear least critical
      localStorage.setItem(CACHE_KEYS[key], JSON.stringify(data));
    }
  };

  const restoreState = (key) => {
    const stored = localStorage.getItem(CACHE_KEYS[key]);
    return stored ? JSON.parse(stored) : null;
  };

  return { isOnline, offlineReady, persistState, restoreState };
}
```

## Simulation Engine

### File: `/src/hooks/useSimulator.js`

```javascript
import { create } from 'zustand';
import { dijkstra, getAlternativeRoutes } from '../logic/router';
import { getSathiMessage } from '../logic/sathiEngine';

export const useMDTStore = create((set, get) => ({
  // State
  graph: null,
  origin: null,
  destination: null,
  activeRoute: null,
  alternativeRoutes: [],
  activeDisruptions: [],
  sathiMessages: [],
  sathiTyping: false,
  language: "hinglish",
  showCrowdHeatmap: false,
  carbonScore: 0,

  // Actions
  setOrigin: (station) => set({ origin: station }),
  setDestination: (station) => set({ destination: station }),

  planRoute: async () => {
    const { graph, origin, destination, activeDisruptions } = get();
    if (!origin || !destination) return;

    set({ sathiTyping: true });

    // Apply disruptions to graph
    const modifiedGraph = applyDisruptions(graph, activeDisruptions);
    
    const routes = getAlternativeRoutes(modifiedGraph, origin.id, destination.id);
    const bestRoute = routes[0]?.route;

    if (bestRoute) {
      const msg = await getSathiMessage({
        eventType: "routeFound",
        route: { ...bestRoute, origin, destination }
      });

      set({
        activeRoute: bestRoute,
        alternativeRoutes: routes.slice(1),
        sathiTyping: false,
        sathiMessages: [...get().sathiMessages, { text: msg, type: "success" }],
        carbonScore: get().carbonScore + (bestRoute.co2Saved || 0)
      });
    }
  },

  toggleDisruption: async (scenario) => {
    const { activeDisruptions, origin, destination } = get();
    const isActive = activeDisruptions.includes(scenario.id);
    const updated = isActive
      ? activeDisruptions.filter(id => id !== scenario.id)
      : [...activeDisruptions, scenario.id];

    set({ activeDisruptions: updated, sathiTyping: true });

    // Wait for dramatic effect
    await delay(800);

    // Add Sathi warning
    const warning = { text: scenario.sathiWarning, type: "warning" };
    set({ sathiMessages: [...get().sathiMessages, warning] });

    // Auto re-route if we have an active route
    if (origin && destination) {
      await get().planRoute();
    }

    set({ sathiTyping: false });

    // Persist disruption state for offline recovery
    localStorage.setItem("mdt_active_disruptions", JSON.stringify(updated));
  }
}));
```

**Implementation:**
```javascript
// /src/logic/monsoonScore.js
export function calculateMonsoonScore(route, nodes) {
  let score = 100;
  route.segments.forEach(seg => {
    const node = nodes[seg.to.id];
    if (node.floodProne) score -= 25;
    if (seg.mode.includes("bus")) score -= 10; // Buses slow in floods
    if (seg.mode.includes("metro")) score += 5; // Metro elevated = safe
  });
  return Math.max(0, Math.min(100, score));
}
```

**Judge Appeal:** Mumbai-specific, deeply relevant, shows local knowledge.

## Appendix B — Common Pitfalls & Fixes

| Problem | Fix |
|---|---|
| Leaflet SSR error | Import with `dynamic(() => import(...), { ssr: false })` |
| localStorage quota exceeded | Clear Sathi log (least critical cache key) |
| Dijkstra infinite loop | Always check `cost > dist[node]` before processing |
| Sathi API timeout during demo | 2-second timeout → auto-fallback to templates |
| Map not re-rendering on route change | Use `key={routeId}` on Polyline components |
| Marathi fonts not loading | Preload Noto Sans Devanagari in `<head>` |
